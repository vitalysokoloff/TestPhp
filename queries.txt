CREATE TABLE `phptest`.`posts` (`userId` INT(20) NOT NULL , `id` INT(20) NOT NULL AUTO_INCREMENT , `title` VARCHAR(100) NOT NULL , `body` VARCHAR(2000) NOT NULL , PRIMARY KEY (`id`));
CREATE TABLE `phptest`.`comments` (`postId` INT(20), `id` INT(20) NOT NULL AUTO_INCREMENT , `name` VARCHAR(100) NOT NULL , `email` VARCHAR(100) NOT NULL , `body` VARCHAR(2000) NOT NULL , PRIMARY KEY (`id`),  FOREIGN KEY (`postId`) REFERENCES `posts` (`id`) ON DELETE SET NULL );
